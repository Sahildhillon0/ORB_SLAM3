# ORB-SLAM3 build dependencies

## Ubuntu (apt)
# From Dockerfile + README prerequisites
sudo apt update
sudo apt install -y \
  build-essential \
  cmake \
  git \
  wget \
  unzip \
  pkg-config \
  nano \
  libeigen3-dev \
  libopencv-dev \
  libglew-dev \
  libboost-all-dev \
  libssl-dev \
  libgtk-3-dev \
  libglfw3-dev \
  libtbb-dev \
  libgl1-mesa-dev \
  libpython3-dev \
  python3-numpy

## Arch Linux (pacman)
# Install packages providing the same dependencies:
sudo pacman -S --needed \
  base-devel \
  cmake \
  git \
  wget \
  unzip \
  pkgconf \
  eigen \
  opencv \
  glew \
  boost \
  openssl \
  gtk3 \
  glfw \
  tbb \
  mesa \
  pangolin \
  python \
  python-numpy

# If pangolin is unavailable in your repos, install from source:
# https://github.com/stevenlovegrove/Pangolin
